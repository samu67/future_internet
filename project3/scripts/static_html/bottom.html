function createNode(latitude, longitude, color, size) { var marker = new google.maps.Marker({ position: {lat: latitude, lng: longitude}, map: map, icon: {path: google.maps.SymbolPath.CIRCLE, fillColor: color, fillOpacity: 1.0, strokeColor: 'black', strokeWeight: 1, scale: size }, draggable: false}); return marker; } function createEdge(slat,slong,dlat,dlong, color, weight) { var flightPlanCoordinates = [ {lat: slat, lng: slong}, {lat: dlat, lng: dlong}]; var flightPath = new google.maps.Polyline({ path: flightPlanCoordinates, geodesic: true, strokeColor: color, strokeOpacity: 1.0, strokeWeight: weight, }); flightPath.setMap(map); } function createEdgeFiber(slat,slong,dlat,dlong, color, weight) { var flightPlanCoordinates = [ {lat: slat, lng: slong}, {lat: dlat, lng: dlong}]; var lineSymbol = { path: 'M 0,-1 0,1', strokeOpacity: 1, scale: 4.5 }; var flightPath = new google.maps.Polyline({ path: flightPlanCoordinates, strokeOpacity: 0, icons: [{ icon: lineSymbol, offset: '0', repeat: '16px' }], }); flightPath.setMap(map); } }
